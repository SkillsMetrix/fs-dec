public sealed abstract class Payment
        permits UpiPayment, CardPayment, CashPayment {

    public abstract String paymentId();
    public abstract String userId();
    public abstract double amount();
}


--

public final record UpiPayment(
        String paymentId,
        String userId,
        double amount,
        String upiId
) extends Payment {}


-----


public final record CardPayment(
        String paymentId,
        String userId,
        double amount,
        String cardNumber
) extends Payment {}
